# Minimal kernel config for RISC-V 64 (QEMU sifive_u + real HiFive Unmatched).
#
# Build:
#   make mrproper && make tinyconfig
#   cat kernel_config.txt >> .config
#   make ARCH=riscv CROSS_COMPILE=riscv64-unknown-linux-gnu- olddefconfig
#   make ARCH=riscv CROSS_COMPILE=riscv64-unknown-linux-gnu- -j$(nproc)

# RISC-V 64-bit
CONFIG_64BIT=y
CONFIG_ARCH_RV64I=y
CONFIG_RISCV_ISA_C=y
CONFIG_RISCV_SBI=y
CONFIG_RISCV_SBI_V01=y

# SMP (FU740 has 5 harts)
CONFIG_SMP=y
CONFIG_NR_CPUS=8

# Console / TTY
CONFIG_TTY=y
CONFIG_PRINTK=y

# SiFive UART driver (ttySIF0)
CONFIG_SERIAL_SIFIVE=y
CONFIG_SERIAL_SIFIVE_CONSOLE=y
CONFIG_SERIAL_EARLYCON=y
CONFIG_SERIAL_EARLYCON_RISCV_SBI=y
CONFIG_HVC_RISCV_SBI=y

# Also NS16550 for QEMU virt fallback
CONFIG_SERIAL_8250=y
CONFIG_SERIAL_8250_CONSOLE=y
CONFIG_SERIAL_OF_PLATFORM=y

# Initramfs (u-root CPIO)
CONFIG_BLK_DEV_INITRD=y
CONFIG_BLK_DEV_RAM=y

# Required for Go runtime
CONFIG_BINFMT_ELF=y
CONFIG_FUTEX=y
CONFIG_EPOLL=y

# Device / filesystem basics
CONFIG_DEVTMPFS=y
CONFIG_DEVTMPFS_MOUNT=y
CONFIG_PROC_FS=y
CONFIG_SYSFS=y
CONFIG_TMPFS=y

# Device tree (required for RISC-V)
CONFIG_OF=y
CONFIG_OF_EARLY_FLATTREE=y

# Timer
CONFIG_RISCV_TIMER=y
CONFIG_HZ_100=y

# Memory
CONFIG_MMU=y
CONFIG_SPARSEMEM=y
CONFIG_PAGE_OFFSET=0xffffaf8000000000

# Virtio (for QEMU)
CONFIG_VIRTIO=y
CONFIG_VIRTIO_MMIO=y
CONFIG_VIRTIO_CONSOLE=y

# vfat filesystem
CONFIG_BLOCK=y
CONFIG_MSDOS_FS=y
CONFIG_VFAT_FS=y
CONFIG_FAT_DEFAULT_CODEPAGE=437
CONFIG_FAT_DEFAULT_IOCHARSET="iso8859-1"
CONFIG_NLS=y
CONFIG_NLS_CODEPAGE_437=y
CONFIG_NLS_ISO8859_1=y

# 9P filesystem
CONFIG_NET_9P=y
CONFIG_NET_9P_VIRTIO=y
CONFIG_9P_FS=y

# Loop + squashfs
CONFIG_BLK_DEV_LOOP=y
CONFIG_MISC_FILESYSTEMS=y
CONFIG_SQUASHFS=y

# Kexec support
CONFIG_CRYPTO=y
CONFIG_KEXEC=y
